# Verify location
pwd
ls

# Check git state
git status
git branch

# Get latest from GitHub
git pull

# Preview a patch
git apply --stat my-change.patch
git apply --check my-change.patch

# Apply a patch
git apply my-change.patch

# Undo a patch
git apply -R my-change.patch

# See what changed
git diff

# Stage changes
git add app/calculator/page.tsx
git add .          # stage all

# Commit
git commit -m "Short description"

# Push to GitHub
git push

# Run app
npm run dev

# Inspect last commit
git show

# File history
git log app/calculator/page.tsx

# Discard local changes
git restore app/calculator/page.tsx

# Hard reset (danger)
git reset --hard
git pull

# Delete untracked file (Windows)
del my-change.patch
